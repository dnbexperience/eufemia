### Signal button

Medium is equivalent to 24, but responsive. To import custom icons, use: `import { bell_medium as Bell } from '@dnb/eufemia/icons'`

<!-- prettier-ignore -->
<ComponentBox scope={{ Bell }}>
	{
	() => /* jsx */ `
<Button
  variant="signal"
  text="Signal Button"
  icon={Bell}
  data-visual-test="button-signal"
/>
	`
	}
</ComponentBox>
