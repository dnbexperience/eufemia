### Horizontal and vertical slider in sync with input field

<ComponentBox useRender>
  {/* jsx */ `
const Component = () => {
  const [value, setValue] = React.useState(70)
  return (<>
    <Slider
      value={value}
      step={10}
      hide_buttons="true"
      label="Slider A:"
      on_change={({ value }) => setValue(value)}
    />
    <VerticalWrapper>
      <Slider
        value={value}
        vertical={true}
        hide_buttons={true}
        use_scrollwheel={true}
        step={1}
        label="Slider B:"
        label_direction="vertical"
        on_change={({ value }) => setValue(value)}
      />
      <Input
        align="center"
        selectall
        value={String(value)}
        on_change={({ value }) => setValue(value)}
      />
    </VerticalWrapper>
  </>)
}
const VerticalWrapper = styled.div\`
  display: inline-flex;
  flex-direction: column;
  align-items: center;
  height: 20rem;/* max-height works fine except in Safari */
  margin-top: 1rem;
  background: rgba(0,0,0,0.1);
  .dnb-input {
    width: 4rem;
    margin-top: 1rem;
  }
\`
render(<Component />)
  `}
</ComponentBox>
