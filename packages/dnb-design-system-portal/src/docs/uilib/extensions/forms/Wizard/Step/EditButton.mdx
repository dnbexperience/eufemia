# NextButton

## Description

`Wizard.EditButton` is a button to be placed in a summary.

```jsx
import { Hr } from '@dnb/eufemia'
import { Form, Wizard, Value } from '@dnb/eufemia/extensions/forms'

render(
  <Form.Handler>
    <Wizard.Layout>
      <Wizard.Step title="Step 1">
        <Card stack>
          <Value.String label="First name" path="/firstName" />
          <Hr />
          <Wizard.EditButton />
        </Card>
      </Wizard.Step>
    </Wizard.Layout>
  </Form.Handler>,
)
```
