## Description

`StepsLayout.useStep` is a React Hook that connects to the `StepsContext` to handle step related operations.

```jsx
import { StepsLayout, Form } from '@dnb/eufemia/extensions/forms'

function Step1() {
  const { activeIndex, setActiveIndex } = StepsLayout.useStep()

  return <StepsLayout.Step>...</StepsLayout.Step>
}

function MyForm() {
  return (
    <Form.Handler>
      <StepsLayout>
        <Step1 />
      </StepsLayout>
    </Form.Handler>
  )
}
```

You can also connect the hook with the `StepsLayout` via an `id` (string). This lets you render the hook outside of the context:

```jsx
import { StepsLayout, Form } from '@dnb/eufemia/extensions/forms'

function Sidecar() {
  const { activeIndex, setActiveIndex } = StepsLayout.useStep('unique-id')
}

function MyForm() {
  return (
    <Form.Handler>
      <Sidecar />
      <StepsLayout id="unique-id">...</StepsLayout>
    </Form.Handler>
  )
}
```
