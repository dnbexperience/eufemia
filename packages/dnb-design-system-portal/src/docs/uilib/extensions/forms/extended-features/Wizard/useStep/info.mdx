## Description

`Wizard.useStep` is a React Hook that connects to the `Wizard.Context` to handle step related operations.

```jsx
import { Form, Wizard } from '@dnb/eufemia/extensions/forms'

function Step1() {
  const { activeIndex, setActiveIndex } = Wizard.useStep()

  return <Wizard.Step>...</Wizard.Step>
}

function MyForm() {
  return (
    <Form.Handler>
      <Wizard.Container>
        <Step1 />
      </Wizard.Container>
    </Form.Handler>
  )
}
```

You can also connect the hook with the `Wizard.Container` via an `id` (string). This lets you render the hook outside of the context:

```jsx
import { Form } from '@dnb/eufemia/extensions/forms'

function Sidecar() {
  const { activeIndex, setActiveIndex } = Wizard.useStep('unique-id')
}

function MyForm() {
  return (
    <Form.Handler>
      <Sidecar />
      <Wizard.Container id="unique-id">...</Wizard.Container>
    </Form.Handler>
  )
}
```
